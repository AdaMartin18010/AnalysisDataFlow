挑战3: 状态管理设计

设计高效的状态管理方案，实现 Exactly-Once 语义

📝 挑战描述

高级

在大规模流处理作业中，状态管理是核心挑战之一。本挑战要求你设计一个高效、可靠的状态管理方案，处理用户行为去重和会话分析场景。

业务场景

实时用户行为分析系统，需要：

  • 24小时内用户行为去重 (使用 ValueState)
  • 用户会话分析 (使用 MapState 存储会话属性)
  • 状态 TTL 自动清理过期数据 (24小时)
  • 使用 RocksDB 状态后端处理大规模状态
  • 实现作业重启后的状态恢复
  • 监控状态大小和访问延迟

完整实现代码

import org.apache.flink.api.common.state.*;
import org.apache.flink.api.common.time.Time;
import org.apache.flink.configuration.Configuration;
import org.apache.flink.contrib.streaming.state.EmbeddedRocksDBStateBackend;
import org.apache.flink.streaming.api.datastream.DataStream;
import org.apache.flink.streaming.api.environment.StreamExecutionEnvironment;
import org.apache.flink.streaming.api.functions.KeyedProcessFunction;
import org.apache.flink.util.Collector;

public class StateManagementChallenge {
    
    public static void main(String[] args) throws Exception {
        StreamExecutionEnvironment env = 
            StreamExecutionEnvironment.getExecutionEnvironment();
        
        // 配置 RocksDB 状态后端
        EmbeddedRocksDBStateBackend rocksDbBackend = 
            new EmbeddedRocksDBStateBackend(true);
        env.setStateBackend(rocksDbBackend);
        env.enableCheckpointing(60000);
        
        // 模拟数据
        DataStream<UserEvent> events = env.fromElements(
            new UserEvent("user1", "evt-001", "/home", "view", 1000),
            new UserEvent("user1", "evt-001", "/home", "view", 2000),  // 重复
            new UserEvent("user1", "evt-002", "/product", "view", 61000)
        );
        
        DataStream<SessionResult> result = events
            .keyBy(UserEvent::getUserId)
            .process(new DeduplicateAndAnalyzeFunction());
        
        result.print();
        env.execute();
    }
    
    public static class DeduplicateAndAnalyzeFunction 
        extends KeyedProcessFunction<String, UserEvent, SessionResult> {
        
        private static final long SESSION_TIMEOUT = 30 * 60 * 1000; // 30分钟
        
        private ValueState<Boolean> seenState;
        private MapState<String, Integer> pageViewState;
        private ValueState<Long> sessionStartState;
        private ValueState<Long> lastActivityState;
        
        @Override
        public void open(Configuration parameters) {
            // TTL 配置: 24小时
            StateTtlConfig ttlConfig = StateTtlConfig
                .newBuilder(Time.hours(24))
                .setUpdateType(StateTtlConfig.UpdateType.OnCreateAndWrite)
                .setStateVisibility(StateTtlConfig.StateVisibility.NeverReturnExpired)
                .cleanupFullSnapshot()
                .build();
            
            // 去重状态
            ValueStateDescriptor<Boolean> seenDescriptor = 
                new ValueStateDescriptor<>("seen", Boolean.class);
            seenDescriptor.enableTimeToLive(ttlConfig);
            seenState = getRuntimeContext().getState(seenDescriptor);
            
            // 页面访问状态
            MapStateDescriptor<String, Integer> pageViewDescriptor = 
                new MapStateDescriptor<>("pageViews", String.class, Integer.class);
            pageViewDescriptor.enableTimeToLive(ttlConfig);
            pageViewState = getRuntimeContext().getMapState(pageViewDescriptor);
            
            // 会话时间状态
            ValueStateDescriptor<Long> sessionStartDescriptor = 
                new ValueStateDescriptor<>("sessionStart", Long.class);
            sessionStartDescriptor.enableTimeToLive(ttlConfig);
            sessionStartState = getRuntimeContext().getState(sessionStartDescriptor);
            
            ValueStateDescriptor<Long> lastActivityDescriptor = 
                new ValueStateDescriptor<>("lastActivity", Long.class);
            lastActivityDescriptor.enableTimeToLive(ttlConfig);
            lastActivityState = getRuntimeContext().getState(lastActivityDescriptor);
        }
        
        @Override
        public void processElement(UserEvent event, Context ctx, 
                                  Collector<SessionResult> out) throws Exception {
            String eventKey = event.userId + ":" + event.eventId;
            
            // 去重检查
            if (seenState.value() != null) {
                return; // 重复事件，忽略
            }
            seenState.update(true);
            
            // 检查会话超时
            Long lastActivity = lastActivityState.value();
            boolean isNewSession = false;
            
            if (lastActivity == null || 
                event.timestamp - lastActivity > SESSION_TIMEOUT) {
                // 新会话
                isNewSession = true;
                sessionStartState.update(event.timestamp);
                pageViewState.clear();
            }
            
            // 更新页面访问统计
            String pageKey = event.pageUrl;
            Integer count = pageViewState.get(pageKey);
            if (count == null) count = 0;
            pageViewState.put(pageKey, count + 1);
            
            // 更新活动时间
            lastActivityState.update(event.timestamp);
            
            // 计算会话统计
            int totalViews = 0;
            int uniquePages = 0;
            for (Integer c : pageViewState.values()) {
                totalViews += c;
                uniquePages++;
            }
            
            Long sessionStart = sessionStartState.value();
            long duration = sessionStart != null ? 
                event.timestamp - sessionStart : 0;
            
            // 注册会话超时定时器
            ctx.timerService().registerEventTimeTimer(
                event.timestamp + SESSION_TIMEOUT
            );
            
            // 输出结果
            out.collect(new SessionResult(
                event.userId,
                totalViews,
                uniquePages,
                duration,
                isNewSession
            ));
        }
        
        @Override
        public void onTimer(long timestamp, OnTimerContext ctx, 
                          Collector<SessionResult> out) throws Exception {
            // 会话超时，清理状态
            Long lastActivity = lastActivityState.value();
            if (lastActivity != null && timestamp >= lastActivity + SESSION_TIMEOUT) {
                pageViewState.clear();
                sessionStartState.clear();
                lastActivityState.clear();
            }
        }
    }
    
    public static class UserEvent {
        String userId;
        String eventId;
        String pageUrl;
        String eventType;
        long timestamp;
        
        UserEvent() {}
        
        UserEvent(String userId, String eventId, String pageUrl, 
                  String eventType, long timestamp) {
            this.userId = userId;
            this.eventId = eventId;
            this.pageUrl = pageUrl;
            this.eventType = eventType;
            this.timestamp = timestamp;
        }
        
        String getUserId() { return userId; }
    }
    
    public static class SessionResult {
        String userId;
        int pageViewCount;
        int uniquePageCount;
        long sessionDuration;
        boolean isNewSession;
        
        SessionResult(String userId, int pageViewCount, int uniquePageCount,
                     long sessionDuration, boolean isNewSession) {
            this.userId = userId;
            this.pageViewCount = pageViewCount;
            this.uniquePageCount = uniquePageCount;
            this.sessionDuration = sessionDuration;
            this.isNewSession = isNewSession;
        }
        
        @Override
        public String toString() {
            return String.format(
                "SessionResult{user='%s', views=%d, unique=%d, duration=%dms, new=%b}",
                userId, pageViewCount, uniquePageCount, sessionDuration, isNewSession
            );
        }
    }
}
