4.1 Table API基础入门

4.1.1 Table API 概述

Table API 是 Flink 提供的声明式编程接口，具有以下特点：

  • 统一批流处理: 同一套API处理有界和无界数据
  • 自动优化: 内置查询优化器，生成高效执行计划
  • 类型安全: 编译时类型检查，减少运行时错误
  • 与SQL互操作: Table API 和 SQL 可无缝混合使用

4.1.2 环境创建与基本操作

import org.apache.flink.streaming.api.environment.StreamExecutionEnvironment;
import org.apache.flink.table.api.Table;
import org.apache.flink.table.api.bridge.java.StreamTableEnvironment;

public class TableApiBasics {
    public static void main(String[] args) throws Exception {
        // 1. 创建 StreamExecutionEnvironment
        StreamExecutionEnvironment env = 
            StreamExecutionEnvironment.getExecutionEnvironment();
        
        // 2. 创建 StreamTableEnvironment
        StreamTableEnvironment tableEnv = StreamTableEnvironment.create(env);
        
        // 3. 从 DataStream 创建 Table
        DataStream<UserEvent> stream = env.addSource(new UserEventSource());
        Table table = tableEnv.fromDataStream(stream);
        
        // 4. 使用 Table API 进行查询
        Table result = table
            .groupBy($("userId"))
            .select($("userId"), $("amount").sum().as("total_amount"));
        
        // 5. 转换回 DataStream
        DataStream<Row> resultStream = tableEnv.toDataStream(result);
        resultStream.print();
        
        env.execute("Table API Basics");
    }
}

4.2.3 SQL 窗口函数

// TUMBLE 窗口 (滚动窗口)
tableEnv.sqlQuery("""
    SELECT 
        user_id,
        TUMBLE_START(event_time, INTERVAL '10' MINUTE) as window_start,
        TUMBLE_END(event_time, INTERVAL '10' MINUTE) as window_end,
        COUNT(*) as event_count,
        SUM(amount) as total_amount
    FROM user_events
    GROUP BY 
        user_id,
        TUMBLE(event_time, INTERVAL '10' MINUTE)
""");

// HOP 窗口 (滑动窗口)
tableEnv.sqlQuery("""
    SELECT 
        user_id,
        SUM(amount) as total_amount
    FROM user_events
    GROUP BY 
        user_id,
        HOP(event_time, INTERVAL '5' MINUTE, INTERVAL '15' MINUTE)
""");

// SESSION 窗口 (会话窗口)
tableEnv.sqlQuery("""
    SELECT 
        user_id,
        SESSION_START(event_time, INTERVAL '30' MINUTE) as session_start,
        COUNT(*) as events_in_session
    FROM user_events
    GROUP BY 
        user_id,
        SESSION(event_time, INTERVAL '30' MINUTE)
""");

4.3.1 Regular Join (常规 Join)

Regular Join 是无界的，需要为两个输入流维护状态，适合小规模维度表。

// 创建订单流表
tableEnv.executeSql("""
    CREATE TABLE orders (
        order_id STRING,
        user_id STRING,
        product_id STRING,
        order_time TIMESTAMP(3),
        amount DECIMAL(10, 2),
        WATERMARK FOR order_time AS order_time - INTERVAL '5' SECOND
    ) WITH (...)
""");

// 创建用户维度表
tableEnv.executeSql("""
    CREATE TABLE users (
        user_id STRING,
        user_name STRING,
        user_level STRING,
        PRIMARY KEY (user_id) NOT ENFORCED
    ) WITH (...)
""");

// Regular Join - 关联订单和用户信息
tableEnv.sqlQuery("""
    SELECT 
        o.order_id,
        o.amount,
        u.user_name,
        u.user_level
    FROM orders o
    JOIN users u ON o.user_id = u.user_id
""");

4.3.2 Interval Join (时间区间 Join)

Interval Join 限制 Join 的时间范围，避免无限状态增长。

// Interval Join - 关联订单和配送事件
tableEnv.executeSql("""
    CREATE TABLE shipments (
        order_id STRING,
        ship_time TIMESTAMP(3),
        carrier STRING,
        WATERMARK FOR ship_time AS ship_time - INTERVAL '5' SECOND
    ) WITH (...)
""");

// 订单创建后 4 小时内发生的配送事件
tableEnv.sqlQuery("""
    SELECT 
        o.order_id,
        o.amount,
        s.carrier,
        TIMESTAMPDIFF(HOUR, o.order_time, s.ship_time) as ship_delay_hours
    FROM orders o
    JOIN shipments s ON o.order_id = s.order_id
    WHERE o.order_time BETWEEN s.ship_time - INTERVAL '4' HOUR AND s.ship_time
""");

4.3.3 Temporal Join (时态表 Join)

Temporal Join 用于关联版本化的维度数据，根据事件时间获取对应版本。

// 创建汇率时态表 (带版本信息)
tableEnv.executeSql("""
    CREATE TABLE currency_rates (
        currency STRING,
        rate DECIMAL(10, 6),
        update_time TIMESTAMP(3),
        WATERMARK FOR update_time AS update_time - INTERVAL '1' SECOND,
        PRIMARY KEY (currency, update_time) NOT ENFORCED
    ) WITH (...)
""");

// 创建带货币的订单表
tableEnv.executeSql("""
    CREATE TABLE orders_with_currency (
        order_id STRING,
        currency STRING,
        amount DECIMAL(10, 2),
        order_time TIMESTAMP(3),
        WATERMARK FOR order_time AS order_time - INTERVAL '5' SECOND
    ) WITH (...)
""");

// Temporal Join - 获取订单发生时的汇率
tableEnv.sqlQuery("""
    SELECT 
        o.order_id,
        o.currency,
        o.amount,
        r.rate,
        o.amount * r.rate as amount_usd,
        o.order_time
    FROM orders_with_currency o
    JOIN currency_rates FOR SYSTEM_TIME AS OF o.order_time r
    ON o.currency = r.currency
""");

4.3.4 复杂查询示例

// 多表关联 + 窗口聚合
tableEnv.sqlQuery("""
    WITH hourly_stats AS (
        SELECT 
            u.user_level,
            TUMBLE_START(o.order_time, INTERVAL '1' HOUR) as hour_start,
            COUNT(*) as order_count,
            SUM(o.amount) as total_amount,
            AVG(o.amount) as avg_amount
        FROM orders o
        JOIN users u ON o.user_id = u.user_id
        GROUP BY 
            u.user_level,
            TUMBLE(o.order_time, INTERVAL '1' HOUR)
    )
    SELECT 
        user_level,
        hour_start,
        order_count,
        total_amount,
        avg_amount,
        RANK() OVER (PARTITION BY hour_start ORDER BY total_amount DESC) as rank
    FROM hourly_stats
""");

// Top-N 查询 (每类商品销量前3)
tableEnv.sqlQuery("""
    SELECT * FROM (
        SELECT 
            product_id,
            COUNT(*) as sales_count,
            ROW_NUMBER() OVER (PARTITION BY category ORDER BY COUNT(*) DESC) as rn
        FROM orders
        GROUP BY product_id, category
    ) WHERE rn <= 3
""");

4.4.3 自定义函数 (UDF)

// Scalar Function - 标量函数
public static class MaskFunction extends ScalarFunction {
    public String eval(String str) {
        if (str == null || str.length() < 4) {
            return "****";
        }
        return str.substring(0, 2) + "****" + str.substring(str.length() - 2);
    }
}

// 注册并使用
tableEnv.createTemporarySystemFunction("mask", MaskFunction.class);
tableEnv.sqlQuery("SELECT mask(phone_number) FROM users");

// Table Function - 表函数 (一对多)
public static class SplitFunction extends TableFunction<Row> {
    public void eval(String str) {
        for (String s : str.split(",")) {
            collect(Row.of(s.trim()));
        }
    }
    
    @Override
    public TypeInformation<Row> getResultType() {
        return Types.ROW(Types.STRING);
    }
}

// 使用 Table Function
tableEnv.sqlQuery("""
    SELECT u.user_id, t.tag
    FROM users u,
    LATERAL TABLE(split_tags(u.tags)) AS t(tag)
""");

// Aggregate Function - 聚合函数
public static class WeightedAvg extends AggregateFunction<Double, WeightedAvgAccum> {
    @Override
    public WeightedAvgAccum createAccumulator() {
        return new WeightedAvgAccum();
    }
    
    public void accumulate(WeightedAvgAccum acc, Double value, Long weight) {
        acc.sum += value * weight;
        acc.count += weight;
    }
    
    @Override
    public Double getValue(WeightedAvgAccum acc) {
        return acc.count == 0 ? 0.0 : acc.sum / acc.count;
    }
}

public static class WeightedAvgAccum {
    public double sum = 0;
    public long count = 0;
}
