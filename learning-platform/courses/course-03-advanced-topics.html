3.1 Checkpoint与容错

🎯 学习目标

  • 深入理解 Checkpoint 机制

3.1.1 Checkpoint 配置

// Checkpoint 基础配置
CheckpointConfig checkpointConfig = env.getCheckpointConfig();

// 启用 Checkpoint，每10秒触发一次
env.enableCheckpointing(10000, CheckpointingMode.EXACTLY_ONCE);

// Checkpoint 存储
checkpointConfig.setCheckpointStorage(new FileSystemCheckpointStorage(
    "hdfs://namenode:8020/flink/checkpoints"
));

// 高级配置
checkpointConfig.setMinPauseBetweenCheckpoints(5000);     // 最小间隔5秒
checkpointConfig.setCheckpointTimeout(60000);             // 超时1分钟
checkpointConfig.setMaxConcurrentCheckpoints(1);          // 最大并发1
checkpointConfig.enableExternalizedCheckpoints(
    ExternalizedCheckpointCleanup.RETAIN_ON_CANCELLATION  // 取消时保留
);
checkpointConfig.enableUnalignedCheckpoints();            // 启用非对齐
checkpointConfig.setAlignmentTimeout(Duration.ofSeconds(30)); // 对齐超时

3.1.2 非对齐 Checkpoint (Unaligned Checkpoint)

💡 提示: Flink 1.11+ 引入了非对齐 Checkpoint，在反压场景下仍能快速完成 Checkpoint，避免 Checkpoint 超时。 
// 对齐 vs 非对齐 Checkpoint 对比

// 对齐 Checkpoint (传统方式)
// 1. Barrier 从 Source 向下游传播
// 2. 每个算子等待所有输入通道的 Barrier 到达
// 3. 阻塞数据流，进行状态快照
// 缺点: 反压时 Barrier 传播慢，导致 Checkpoint 超时

// 非对齐 Checkpoint (推荐)
checkpointConfig.enableUnalignedCheckpoints(true);
checkpointConfig.setAlignmentTimeout(Duration.ofSeconds(10));
// 超过10秒未完成对齐，自动切换到非对齐模式
// 非对齐模式将数据缓存到 Checkpoint 中，Barrier 可以跳过积压的数据

3.1.3 增量 Checkpoint

// RocksDB 增量 Checkpoint 配置
EmbeddedRocksDBStateBackend rocksDbBackend = 
    new EmbeddedRocksDBStateBackend(true);  // true = 增量

// 增量 Checkpoint 参数调优
Configuration config = new Configuration();
config.setString("state.backend.incremental", "true");
config.setString("state.backend.rocksdb.memory.managed", "true");
config.setString("state.backend.rocksdb.checkpoint.transfer.thread.num", "4");
env.configure(config);

// 监控 Checkpoint 指标
// - checkpointDuration: Checkpoint 耗时
// - checkpointedSize: Checkpoint 数据大小
// - numberOfFailedCheckpoints: 失败次数
⚠️ 注意: 增量 Checkpoint 虽然减少了数据量，但会增加恢复时间（需要按顺序重放增量）。对于超大状态，需要权衡 Checkpoint 频率和恢复时间。

3.2.1 背压检测

// 背压指标监控 (通过 REST API)
// GET /jobs/{jobId}/vertices/{vertexId}/backpressure

// 背压状态
// OK: 无背压 (比值 0-0.1)
// LOW: 轻度背压 (比值 0.1-0.5)
// HIGH: 严重背压 (比值 > 0.5)

// 代码中检测背压
public class BackpressureMonitor extends RichMapFunction<Event, Event> {
    private transient Meter outputRate;
    private transient Histogram latency;
    
    @Override
    public void open(Configuration parameters) {
        outputRate = getRuntimeContext()
            .getMetricGroup()
            .meter("outputRate", new MeterView(60));
        latency = getRuntimeContext()
            .getMetricGroup()
            .histogram("eventLatency", new DropwizardHistogramWrapper(
                new com.codahale.metrics.Histogram(
                    new SlidingWindowReservoir(500)
                )
            ));
    }
    
    @Override
    public Event map(Event event) {
        outputRate.markEvent();
        latency.update(System.currentTimeMillis() - event.getTimestamp());
        return event;
    }
}

3.2.2 自适应限流

// 基于 Buffer 膨胀率的自适应限流
public class AdaptiveThrottler extends RichMapFunction<Event, Event> {
    private transient double currentLimit;
    private static final double MAX_LIMIT = 10000;
    private static final double MIN_LIMIT = 100;
    
    @Override
    public void open(Configuration parameters) {
        currentLimit = MAX_LIMIT;
        
        // 注册指标
        getRuntimeContext().getMetricGroup()
            .gauge("throttleLimit", () -> currentLimit);
    }
    
    @Override
    public Event map(Event event) throws Exception {
        // 获取当前 Buffer 膨胀率 (通过系统指标)
        double bufferUsage = getBufferUsage();
        
        if (bufferUsage > 0.8) {
            // 高负载，降低限流阈值
            currentLimit = Math.max(MIN_LIMIT, currentLimit * 0.9);
        } else if (bufferUsage < 0.3) {
            // 低负载，提高限流阈值
            currentLimit = Math.min(MAX_LIMIT, currentLimit * 1.1);
        }
        
        // 根据当前限流阈值决定是否处理
        if (shouldProcess(currentLimit)) {
            return event;
        } else {
            // 丢弃或采样
            return null;
        }
    }
}

3.2.3 异步 IO 优化

// 使用 AsyncFunction 解决 IO 阻塞导致的背压
public class AsyncDatabaseRequest 
    extends RichAsyncFunction<String, EnrichedEvent> {
    
    private transient ExecutorService executor;
    private transient DatabaseClient client;
    
    @Override
    public void open(Configuration parameters) {
        executor = Executors.newFixedThreadPool(100);
        client = new DatabaseClient();
    }
    
    @Override
    public void asyncInvoke(String key, ResultFuture<EnrichedEvent> resultFuture) {
        CompletableFuture.supplyAsync(() -> client.query(key), executor)
            .thenAccept(result -> {
                resultFuture.complete(Collections.singletonList(
                    new EnrichedEvent(key, result)
                ));
            })
            .exceptionally(throwable -> {
                resultFuture.completeExceptionally(throwable);
                return null;
            });
    }
    
    @Override
    public void timeout(String key, ResultFuture<EnrichedEvent> resultFuture) {
        resultFuture.complete(Collections.singletonList(
            new EnrichedEvent(key, null)  // 降级处理
        ));
    }
}

// 使用异步 IO
DataStream<EnrichedEvent> enriched = AsyncDataStream.unorderedWait(
    inputStream,
    new AsyncDatabaseRequest(),
    1000,    // 超时 1秒
    TimeUnit.MILLISECONDS,
    100      // 最大并发请求数
);

3.3.4 热点解决

// 两阶段聚合解决热点
DataStream<Result> result = stream
    // 第一阶段: 随机预聚合
    .map(new RichMapFunction<Event, Tuple2<String, Long>>() {
        private Random random;
        
        @Override
        public void open(Configuration parameters) {
            random = new Random();
        }
        
        @Override
        public Tuple2<String, Long> map(Event event) {
            // 添加随机前缀分散热点
            String prefixedKey = random.nextInt(100) + "_" + event.getKey();
            return Tuple2.of(prefixedKey, event.getValue());
        }
    })
    .keyBy(t -> t.f0)
    .window(TumblingEventTimeWindows.of(Time.seconds(10)))
    .aggregate(new SumAggregate())
    // 第二阶段: 去除前缀后最终聚合
    .map(t -> {
        String originalKey = t.f0.substring(t.f0.indexOf("_") + 1);
        return Tuple2.of(originalKey, t.f1);
    })
    .keyBy(t -> t.f0)
    .window(TumblingEventTimeWindows.of(Time.seconds(10)))
    .aggregate(new FinalSumAggregate());

3.4.3 监控集成

// Prometheus 指标导出
# flink-conf.yaml
metrics.reporters: prom
metrics.reporter.prom.class: org.apache.flink.metrics.prometheus.PrometheusReporter
metrics.reporter.prom.port: 9249

// 自定义指标
public class MonitoredFunction extends RichMapFunction<Event, Event> {
    private transient Counter eventCounter;
    private transient Histogram processingTime;
    
    @Override
    public void open(Configuration parameters) {
        MetricGroup group = getRuntimeContext().getMetricGroup();
        eventCounter = group.counter("eventsProcessed");
        processingTime = group.histogram("processingTimeNs", 
            new DropwizardHistogramWrapper(
                new Histogram(new SlidingWindowReservoir(500))));
    }
    
    @Override
    public Event map(Event event) {
        long start = System.nanoTime();
        Event result = process(event);
        eventCounter.inc();
        processingTime.update(System.nanoTime() - start);
        return result;
    }
}
