2.1.1 Session Window 深入

Session Window 根据活动间隙动态划分窗口，非常适合分析用户行为会话。

// 动态会话窗口 (间隙由数据决定)
DataStream<Result> dynamicSession = events
    .keyBy(Event::getUserId)
    .window(new DynamicSessionWindow(Time.minutes(30)))
    .allowedLateness(Time.minutes(10))
    .sideOutputLateData(lateDataTag)
    .process(new SessionProcessFunction());

// 自定义 Session Window 合并逻辑
public class SessionProcessFunction 
    extends ProcessWindowFunction<Event, Result, String, TimeWindow> {
    
    @Override
    public void process(String key, Context context, 
                       Iterable<Event> elements, Collector<Result> out) {
        long sessionStart = Long.MAX_VALUE;
        long sessionEnd = Long.MIN_VALUE;
        int eventCount = 0;
        
        for (Event e : elements) {
            sessionStart = Math.min(sessionStart, e.getTimestamp());
            sessionEnd = Math.max(sessionEnd, e.getTimestamp());
            eventCount++;
        }
        
        long sessionDuration = sessionEnd - sessionStart;
        out.collect(new Result(key, eventCount, sessionDuration));
    }
}

2.1.2 自定义 Trigger

// 自定义触发器: 当计数达到100时提前触发
public class CountTrigger<T> extends Trigger<T, TimeWindow> {
    private final long maxCount;
    private final ReducingStateDescriptor<Long> stateDesc;
    
    public CountTrigger(long maxCount) {
        this.maxCount = maxCount;
        this.stateDesc = new ReducingStateDescriptor<>(
            "count", Long::sum, LongSerializer.INSTANCE);
    }
    
    @Override
    public TriggerResult onElement(T element, long timestamp, 
                                   TimeWindow window, TriggerContext ctx) throws Exception {
        ReducingState<Long> count = ctx.getPartitionedState(stateDesc);
        count.add(1L);
        
        if (count.get() >= maxCount) {
            count.clear();
            return TriggerResult.FIRE_AND_PURGE;
        }
        return TriggerResult.CONTINUE;
    }
    
    @Override
    public TriggerResult onProcessingTime(long time, TimeWindow window, 
                                         TriggerContext ctx) {
        return TriggerResult.FIRE;
    }
    
    @Override
    public TriggerResult onEventTime(long time, TimeWindow window, 
                                    TriggerContext ctx) {
        return time == window.maxTimestamp() ? 
            TriggerResult.FIRE : TriggerResult.CONTINUE;
    }
}

2.1.3 窗口 Join 操作

// Tumbling Window Join
DataStream<Order> orders = ...;
DataStream<Payment> payments = ...;

DataStream<OrderPayment> joined = orders
    .join(payments)
    .where(Order::getOrderId)
    .equalTo(Payment::getOrderId)
    .window(TumblingEventTimeWindows.of(Time.minutes(5)))
    .apply(new JoinFunction<Order, Payment, OrderPayment>() {
        @Override
        public OrderPayment join(Order order, Payment payment) {
            return new OrderPayment(order, payment);
        }
    });

// Interval Join (基于时间区间的 Join)
DataStream<OrderPayment> intervalJoined = orders
    .keyBy(Order::getOrderId)
    .intervalJoin(payments.keyBy(Payment::getOrderId))
    .between(Time.minutes(-10), Time.minutes(5))  // 订单前10分钟到后5分钟
    .process(new ProcessJoinFunction<Order, Payment, OrderPayment>() {
        @Override
        public void processElement(Order order, Payment payment, 
                                  Context ctx, Collector<OrderPayment> out) {
            out.collect(new OrderPayment(order, payment));
        }
    });

2.2.2 Keyed State 使用

// 使用 ValueState 实现去重
public class DeduplicateFunction 
    extends KeyedProcessFunction<String, Event, Event> {
    
    private ValueState<Boolean> seenState;
    
    @Override
    public void open(Configuration parameters) {
        StateTtlConfig ttlConfig = StateTtlConfig
            .newBuilder(Time.hours(24))
            .setUpdateType(OnCreateAndWrite)
            .setStateVisibility(NeverReturnExpired)
            .cleanupFullSnapshot()
            .build();
        
        ValueStateDescriptor<Boolean> descriptor = 
            new ValueStateDescriptor<>("seen", Boolean.class);
        descriptor.enableTimeToLive(ttlConfig);
        seenState = getRuntimeContext().getState(descriptor);
    }
    
    @Override
    public void processElement(Event event, Context ctx, Collector<Event> out) 
            throws Exception {
        if (seenState.value() == null) {
            seenState.update(true);
            out.collect(event);  // 首次出现，输出
        }
        // 重复事件被丢弃
    }
}

// 使用 MapState 实现用户属性累积
public class UserProfileFunction 
    extends KeyedProcessFunction<String, Event, UserProfile> {
    
    private MapState<String, Integer> attributeCounts;
    
    @Override
    public void open(Configuration parameters) {
        attributeCounts = getRuntimeContext().getMapState(
            new MapStateDescriptor<>("attributes", String.class, Integer.class)
        );
    }
    
    @Override
    public void processElement(Event event, Context ctx, 
                              Collector<UserProfile> out) throws Exception {
        String attr = event.getAttribute();
        Integer count = attributeCounts.get(attr);
        if (count == null) count = 0;
        attributeCounts.put(attr, count + 1);
        
        // 构建用户画像
        UserProfile profile = new UserProfile(ctx.getCurrentKey());
        for (Map.Entry<String, Integer> entry : attributeCounts.entries()) {
            profile.addAttribute(entry.getKey(), entry.getValue());
        }
        out.collect(profile);
    }
}

2.3.1 Pattern API 基础

// 定义模式: 登录失败3次后成功登录 (可能的暴力破解)
Pattern<LoginEvent, ?> pattern = Pattern
    .<LoginEvent>begin("failures")
        .where(new SimpleCondition<LoginEvent>() {
            @Override
            public boolean filter(LoginEvent event) {
                return event.getStatus().equals("FAIL");
            }
        })
        .times(3)                    // 恰好3次
        .within(Time.minutes(5))     // 5分钟内
    .next("success")                // 紧接着
        .where(new SimpleCondition<LoginEvent>() {
            @Override
            public boolean filter(LoginEvent event) {
                return event.getStatus().equals("SUCCESS");
            }
        })
    .within(Time.minutes(10));

// 应用模式
PatternStream<LoginEvent> patternStream = CEP.pattern(
    loginEvents.keyBy(LoginEvent::getUserId), 
    pattern
);

// 处理匹配结果
DataStream<Alert> alerts = patternStream
    .process(new PatternProcessFunction<LoginEvent, Alert>() {
        @Override
        public void processMatch(
            Map<String, List<LoginEvent>> match,
            Context ctx,
            Collector<Alert> out) {
            
            List<LoginEvent> failures = match.get("failures");
            LoginEvent success = match.get("success").get(0);
            
            out.collect(new Alert(
                success.getUserId(),
                "SUSPICIOUS_LOGIN",
                "3 failures followed by success from IP: " + success.getIp()
            ));
        }
    });

2.3.2 高级 Pattern 特性

// 量词修饰符
Pattern<Event, ?> complexPattern = Pattern
    .<Event>begin("start")
        .where(evt -> evt.getType().equals("A"))
    .followedBy("middle")           // 宽松连续 (允许其他事件穿插)
        .where(evt -> evt.getType().equals("B"))
        .oneOrMore()                  // 1次或多次
        .greedy()                     // 贪婪匹配
    .followedByAny("end")           // 从任意分支继续
        .where(evt -> evt.getType().equals("C"))
    .within(Time.seconds(30));

// 迭代条件 (基于之前事件的条件)
Pattern<Event, ?> iterativePattern = Pattern
    .<Event>begin("start")
    .next("middle")
        .where(new IterativeCondition<Event>() {
            @Override
            public boolean filter(Event event, Context<Event> ctx) throws Exception {
                double startTemp = ctx.getEventsForPattern("start")
                    .iterator().next().getTemperature();
                return event.getTemperature() > startTemp + 10;
            }
        })
    .within(Time.minutes(2));

2.3.3 超时处理

// 处理模式匹配超时 (部分匹配)
OutputTag<TimeoutEvent> timeoutTag = new OutputTag<TimeoutEvent>("timeout"){};

Pattern<OrderEvent, ?> orderPattern = Pattern
    .<OrderEvent>begin("create")
        .where(evt -> evt.getType().equals("CREATE"))
    .next("pay")
        .where(evt -> evt.getType().equals("PAY"))
    .within(Time.minutes(30));  // 30分钟内必须支付

SingleOutputStreamOperator<OrderResult> resultStream = CEP.pattern(
    orderEvents.keyBy(OrderEvent::getOrderId),
    orderPattern
).process(new PatternTimeoutHandler());

public class PatternTimeoutHandler 
    extends PatternProcessFunction<OrderEvent, OrderResult>
    implements TimedOutPartialMatchHandler<OrderEvent> {
    
    @Override
    public void processMatch(Map<String, List<OrderEvent>> match, 
                           Context ctx, Collector<OrderResult> out) {
        out.collect(new OrderResult(match.get("create").get(0).getOrderId(), 
                                     "SUCCESS"));
    }
    
    @Override
    public void processTimedOutMatch(Map<String, List<OrderEvent>> match,
                                    Context ctx) {
        OrderEvent create = match.get("create").get(0);
        ctx.output(timeoutTag, new TimeoutEvent(create.getOrderId(), 
                                                   "PAY_TIMEOUT"));
    }
}

// Kafka Source (Flink 1.14+ 新 API)
KafkaSource<String> kafkaSource = KafkaSource
    .<String>builder()
    .setBootstrapServers("kafka:9092")
    .setTopics("input-topic")
    .setGroupId("flink-consumer-group")
    .setStartingOffsets(OffsetsInitializer.earliest())
    .setValueOnlyDeserializer(new SimpleStringSchema())
    .setProperty("isolation.level", "read_committed")  // 读已提交
    .build();

DataStream<String> stream = env.fromSource(
    kafkaSource,
    WatermarkStrategy.forBoundedOutOfOrderness(Duration.ofSeconds(5)),
    "Kafka Source"
);

// Kafka Sink (Exactly-Once)
KafkaSink<String> kafkaSink = KafkaSink
    .<String>builder()
    .setBootstrapServers("kafka:9092")
    .setRecordSerializer(
        KafkaRecordSerializationSchema.builder()
            .setTopic("output-topic")
            .setValueSerializationSchema(new SimpleStringSchema())
            .build()
    )
    .setDeliveryGuarantee(DeliveryGuarantee.EXACTLY_ONCE)
    .setTransactionalIdPrefix("flink-tx-")
    .build();

stream.sinkTo(kafkaSink);

2.4.3 自定义 Source Function

// 自定义并行 Source
public class CustomSource implements ParallelSourceFunction<Event>, 
                                              CheckpointListener {
    
    private volatile boolean running = true;
    private transient ListState<Long> offsetState;
    private long currentOffset = 0;
    
    @Override
    public void run(SourceContext<Event> ctx) throws Exception {
        while (running) {
            synchronized (ctx.getCheckpointLock()) {
                Event event = fetchEvent(currentOffset);
                if (event != null) {
                    ctx.collectWithTimestamp(event, event.getTimestamp());
                    currentOffset++;
                }
            }
            Thread.sleep(100);
        }
    }
    
    @Override
    public void snapshotState(FunctionSnapshotContext context) throws Exception {
        offsetState.update(Collections.singletonList(currentOffset));
    }
    
    @Override
    public void initializeState(FunctionInitializationContext context) throws Exception {
        offsetState = context.getOperatorStateStore().getListState(
            new ListStateDescriptor<>("offset", Long.class)
        );
        if (context.isRestored()) {
            for (Long offset : offsetState.get()) {
                currentOffset = offset;
            }
        }
    }
    
    @Override
    public void cancel() { running = false; }
}
